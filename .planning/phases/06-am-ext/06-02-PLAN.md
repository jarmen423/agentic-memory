**packages/am-ext/adapters/base.js** — Implement BaseAdapter class exactly as specified in CONTEXT.md with these required elements:
- Constructor sets: this.platform, this.selectors, this._debounce_timer=null, this._last_turn_count=0, this._session_id=null, this._turn_index=0, this._observer=null
- start(): extract session ID, _queryFirst for container (silent return if not found), create MutationObserver, call _captureNewTurns() immediately for existing turns
- stop(): if (this._observer) { this._observer.disconnect(); this._observer = null; }
- _onMutation(): clearTimeout + setTimeout 800ms debounce
- _captureNewTurns(): call _extractAllTurns(), slice from _last_turn_count, forEach _sendTurn, update _last_turn_count
- _sendTurn(turn): chrome.runtime.sendMessage({type:"NEW_TURN", payload:{role, content, session_id, platform, turn_index: this._turn_index++}}).catch(() => {})
- _extractSessionId(): window.location.pathname.match(new RegExp(this.selectors.session_id_regex)); fallback to `fallback-${Date.now()}`
- _queryFirst(selector_list): split on comma, map to document.querySelector(s.trim()), find first non-null
- _extractAllTurns(): default implementation queries user_message and assistant_message selectors, interleaves by DOM position (compareDocumentPosition), returns [{role, content}]. Subclasses may override.

**Subclass pattern** for chatgpt.js, claude.js, perplexity.js:
```js
class ChatGPTAdapter extends BaseAdapter {
  _extractAllTurns() {
    const all = [...document.querySelectorAll(
      `${this.selectors.user_message}, ${this.selectors.assistant_message}`
    )].sort((a, b) => a.compareDocumentPosition(b) & Node.DOCUMENT_POSITION_FOLLOWING ? -1 : 1);
    return all.map(el => ({
      role: el.matches(this.selectors.user_message) ? "user" : "assistant",
      content: (el.querySelector(this.selectors.message_content) || el).innerText.trim(),
    }));
  }
}
```

**adapters/gemini.js** — GeminiAdapter must handle custom elements. Override _extractAllTurns():

```js
_extractAllTurns() {
  const all = [...document.querySelectorAll("user-query, model-response")]
    .sort((a, b) => a.compareDocumentPosition(b) & Node.DOCUMENT_POSITION_FOLLOWING ? -1 : 1);
  return all.map(el => {
    const is_user = el.tagName.toLowerCase() === "user-query";
    const content_el = is_user
      ? el.querySelector("div.query-content")
      : el.querySelector("message-content");
    return {
      role: is_user ? "user" : "assistant",
      content: (content_el || el).innerText.trim(),
    };
  });
}
```

```js
// Bundled selectors — inlined at build time (or loaded as a separate content script)
// Since no build step, declare BUNDLED_SELECTORS as a const matching selectors.json content
const BUNDLED_SELECTORS = { ... }; // paste full selectors.json content here

const ADAPTERS = {
  chatgpt: ChatGPTAdapter,
  claude: ClaudeAdapter,
  gemini: GeminiAdapter,
  perplexity: PerplexityAdapter,
};

function detect_platform(hostname) {
  if (hostname.includes("chat.openai.com")) return "chatgpt";
  if (hostname.includes("claude.ai")) return "claude";
  if (hostname.includes("gemini.google.com")) return "gemini";
  if (hostname.includes("perplexity.ai")) return "perplexity";
  return null;
}

(async () => {
  const platform = detect_platform(window.location.hostname);
  if (!platform) return;

  const config = await chrome.storage.sync.get(["endpoint", "platforms", "paused"]);
  if (config.paused) return;
  if (config.platforms && config.platforms[platform] === false) return;

  let selectors = BUNDLED_SELECTORS.platforms[platform];
  try {
    const res = await fetch(`${config.endpoint || "http://localhost:8000"}/ext/selectors.json`);
    if (res.ok) {
      const remote = await res.json();
      if (remote.platforms?.[platform]) selectors = remote.platforms[platform];
    }
  } catch {} // use bundled on failure — intentional silent swallow

  const AdapterClass = ADAPTERS[platform];
  const adapter = new AdapterClass(platform, selectors);
  adapter.start();

  // SPA navigation: detect URL change without page reload
  // Add 200ms delay before restart to avoid race with streaming in-progress (per RESEARCH.md pitfall 4)
  let last_url = window.location.href;
  new MutationObserver(() => {
    if (window.location.href !== last_url) {
      last_url = window.location.href;
      adapter.stop?.();
      setTimeout(() => {
        adapter._session_id = adapter._extractSessionId();
        adapter._last_turn_count = 0;
        adapter._turn_index = 0;
        adapter.start();
      }, 200);
    }
  }).observe(document.body, { subtree: true, childList: true });
})();
```

Note: The 200ms SPA navigation delay is a deliberate fix for RESEARCH.md Pitfall 4 (duplicate turns on navigation during streaming).
