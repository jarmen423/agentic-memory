Create packages/am-ext/tests/utils.test.js using node:test (Node 18+ built-in). No npm dependencies.

**Implementation approach:** Extract the pure logic as standalone functions at the top of the test file (rather than importing from base.js, which uses browser globals). This avoids needing jsdom for every test.

```js
import { test, describe, mock } from 'node:test';
import assert from 'node:assert/strict';

// --- Pure functions extracted from base.js for testing ---

function extract_session_id(pathname, regex_str) {
  const match = pathname.match(new RegExp(regex_str));
  return match ? match[1] : `fallback-${Date.now()}`;
}

function make_debounce(callback, delay_ms) {
  let timer = null;
  return function debounced() {
    clearTimeout(timer);
    timer = setTimeout(callback, delay_ms);
  };
}

function query_first(selector_list, doc) {
  return selector_list.split(',')
    .map(s => doc.querySelector(s.trim()))
    .find(el => el != null) || null;
}

// --- Tests ---

describe('extract_session_id', () => {
  test('chatgpt URL', () => {
    assert.equal(extract_session_id('/c/abc123-def456', '/c/([a-z0-9-]+)'), 'abc123-def456');
  });
  test('claude URL', () => {
    assert.equal(extract_session_id('/chat/xyz789', '/chat/([a-z0-9-]+)'), 'xyz789');
  });
  // ... etc
});
```
